使用 ffmpeg 从视频中提取音频 

# 转 MP3（有损）
ffmpeg -i input.webm -vn -acodec libmp3lame -q:a 2 output.mp3
# -acodec libmp3lame    使用 MP3 编码器 libmp3lame 进行音频编码
# -q:a 2    设置音频质量（质量越低，数字越大；常用范围 0–9）2 表示高质量、文件大小适中

# 转 WAV（无损）
ffmpeg -i input.webm -vn -acodec pcm_s16le output.wav
